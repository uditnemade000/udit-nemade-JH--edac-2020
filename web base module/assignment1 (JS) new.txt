function fact(num)
{
    var f=1;
    for(var i=1;i<=num;i++)
    {
        f*=i;//f=f*i;
    }
    return f;
}
console.log(fact(4));

function amt(amount,coin)
{
  if(amount===0)
    {
      return [];
    }
  else
    {
      if(amount>=coin[0])
        {
          x=(amount-coin[0]);
          return [coin[0]].concat(amt(x,coin));
        }
      else
        {
          coin.shift();
          return amt(amount,coin);
        }
    }
}
